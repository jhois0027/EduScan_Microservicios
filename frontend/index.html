`; const formData = new FormData(); formData.append('file', imagenCapturada); try { console.log('🤖 Enviando a IA:', API_IA_URL); const response = await fetch(`${API_IA_URL}/corregir`, { method: 'POST', body: formData }); if (!response.ok) { throw new Error(`Error HTTP: ${response.status}`); } const correccion = await response.json(); console.log('✅ Respuesta IA:', correccion); // Guardar en base de datos await guardarEvaluacion(alumnoId, correccion); // Mostrar resultados mostrarResultados(correccion); } catch (error) { console.error('❌ Error en corrección:', error); resultadoDiv.innerHTML = `

${error.message}

`; } } async function guardarEvaluacion(alumnoId, correccion) { try { const hoy = new Date().toISOString().split('T')[0]; const evaluacion = { id_alumno: parseInt(alumnoId), puntaje: correccion.nota_final || 0, fecha: hoy, examen_nombre: examenNombre.value, grado: examenGrado.value, respuestas: (correccion.respuestas_alumno || []).map((p, i) => ({ id_pregunta: i + 1, es_correcta: p >= 3.0 })) }; console.log('💾 Guardando evaluación:', evaluacion); const response = await fetch(`${API_DB_URL}/nueva_evaluacion`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(evaluacion) }); if (response.ok) { console.log('✅ Evaluación guardada en BD'); } else { console.warn('⚠️ No se pudo guardar en BD'); } } catch (error) { console.error('❌ Error guardando:', error); } } function mostrarResultados(correccion) { const respuestasHtml = (correccion.respuestas_alumno || []).map((puntaje, idx) => { const esAprobado = puntaje >= 3.0; const clase = esAprobado ? 'correcta' : 'incorrecta'; const icono = esAprobado ? '✅' : '❌'; return `

`; }).join(''); resultadoDiv.innerHTML = `
